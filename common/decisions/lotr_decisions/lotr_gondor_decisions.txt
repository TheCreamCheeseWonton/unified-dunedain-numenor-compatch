			}

			dynasty = { add_dynasty_prestige = 1000 }

			}
			title:e_gondor = {
				change_title_holder = {
					holder = root
					change = scope:change
				}
			}

				every_held_title = {
					add_title_law = male_preference_law # To stop the house of Telcontar from losing the throne immediately if Eldarion dies
				}
			}

			destroy_title = title:e_gondor_steward
