decision_expedition_anarike = {
	decision_group_type = Numenor_major
	title = decision_expedition_anarike_title
	picture = { reference = "gfx/interface/illustrations/decisions/decision_near_harad_01.dds" }
	
	ai_goal = yes
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 100
	}
	
	cost = {
		gold = {
			value = 1000
		}
	}
	
	selection_tooltip = decision_expedition_anarike_tooltip
	desc = decision_expedition_anarike_desc
	confirm_text = decision_expedition_anarike_confirm
	
	is_shown = {
		always = no
		is_landed = yes
		OR = {
			has_primary_title = title:e_numenor
			has_primary_title = title:h_numenor
		}
		# is_target_in_global_variable_list = {
		# 	name = unavailable_unique_decisions
		# 	target = flag:decision_reclaim_thani_hazad_numenor
		# }
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:decision_expedition_anarike
			}
		}
	}
	is_valid = {
		trigger_if = {
			limit = { exists = situation:numenor_situation }
			NOT = { situation:numenor_situation ?= { situation_current_phase = numenor_situation_isolation } }
		}
		completely_controls = title:k_an_balkumagan
		prestige_level >= medium_prestige_level
		OR = {
			has_trait = scholar
			learning > 17
		}
	}
	
	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_available_adult = yes
		is_at_war = no
	}
	
	effect = {
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:decision_expedition_anarike
		}
		custom_tooltip = decision_expedition_anarike
			
		trigger_event = numenor.2001
		save_scope_as = expedition_sender
	}
	ai_will_do = {
		base = 100
	}
}
