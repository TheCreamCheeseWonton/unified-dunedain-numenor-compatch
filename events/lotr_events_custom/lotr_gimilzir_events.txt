gimilzir_events.074 = { # Arrives at Numenor - Complete
	content_source = realms_dlc
	type = character_event
	title = gimilzir_events.074.t
	desc = gimilzir_events.074.desc
	theme = sailing_activity
	override_background = {
		reference = docks
	}
	left_portrait = {
		character = root
		animation = survey
	}
	immediate = {
	play_music_cue = "mx_cue_combat_2"
		hidden_effect = {
			random_list = {
				4 = {
					create_artifact_numenorean_figurine_effect = { OWNER = root }
				}
				2 = {
					create_artifact_numenorean_helm_effect = { OWNER = root }
				}
				2 = {
					create_artifact_numenorean_shield_effect = { OWNER = root }
				}
				1 = {
					create_artifact_numenorean_steelbow_effect = { OWNER = root }
				}
			}
			scope:newly_created_artifact = { save_scope_as = numenorean_artifact }
		}
		add_character_modifier = reached_meneltarma
		add_prestige = 1000
		add_legitimacy_effect = { LEGITIMACY = medium_legitimacy_gain }
		root.dynasty = {
			add_dynasty_prestige = 1000
			add_dynasty_prestige_level = 1
		}
		if = {
			limit = {
				faith_is_good = yes
			}
			add_piety = 750
		}
		if = {
			limit = {
				has_dlc_feature = legends
				dynasty = { NOT = { has_variable = gained_meneltarma_legend } }
			}
			create_legend_seed = {
				type = holy
				quality = illustrious
				chronicle = climbed_the_meneltarma
				properties = {
					dynasty = root.dynasty
					founder = root
					title = root.primary_title.empire
				}
			}
			dynasty = {
				set_variable = {
					name = gained_meneltarma_legend
					value = yes
				}
			}
		}
		province:6146 = {
			set_holding_type = castle_holding
		}
		get_title = title:c_uinen
		title:c_uinen = {
			set_county_faith = root.faith
			set_county_culture = root.culture
			change_development_level = 3

			activate_holy_site = eldalonde_faithful
			activate_holy_site = nindamos_faithful
		}
		every_living_character = { 
			limit = { 
				location ?= { geographical_region = middleearth_numenor }
				is_human = yes
			}
			if = { 
				limit = { has_trait = blood_of_numenor_10 }	
				add_piety = 1000
			}
			else_if = { 
				limit = { has_trait = blood_of_numenor_9 }	
				add_piety = 1000
			}
			else_if = { 
				limit = { has_trait = blood_of_numenor_8 }	
				add_piety = 1000
			}
			else_if = { 
				limit = { has_trait = blood_of_numenor_7 }	
				add_piety = 100
			}
			else_if = { 
				limit = { has_trait = blood_of_numenor_6 }	
				remove_trait = blood_of_numenor_6
				add_trait = blood_of_numenor_7
			}
			else_if = { 
				limit = { has_trait = blood_of_numenor_5 }	
				remove_trait = blood_of_numenor_5
				add_trait = blood_of_numenor_7
			}
			else_if = { 
				limit = { has_trait = blood_of_numenor_4 }

			}
			else_if = {

				add_trait = blood_of_numenor_7
			}
			else_if = {
